import rx.lang.scala.Observable
import scala.concurrent.duration._
import scala.util.Random

object PrimerEjercicio extends App {

  val sensor1 = Observable.interval(1.second)
    .map(_ => Random.between(20, 31))

  val sensor2 = Observable.interval(1500.millis)
    .map(_ => Random.between(30, 41))

  val combined = sensor1.zip(sensor2)

  combined
    .take(10)
    .subscribe(
      onNext = { case (s1, s2) =>
        val diff = Math.abs(s1 - s2)

        println(s"Sensor1: $s1Â°C, Sensor2: $s2Â°C, Diferencia: $diffÂ°C")
